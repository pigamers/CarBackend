Chat System:

// Real-time chat using Socket.io
const initializeChat = (io) => {
  io.on('connection', (socket) => {
    socket.on('join_room', (roomId) => {
      socket.join(roomId);
    });
    
    socket.on('send_message', (data) => {
      io.to(data.roomId).emit('receive_message', data);
    });
  });
};
